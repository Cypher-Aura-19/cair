\section{Hero/Banner Images}
\textit{Large banner images (min 1920x800px) for page headers. Professional quality required.}
\begin{itemize}
    \item Home page -- community gathering or group photo
    \item About page -- diverse Muslim community members
    \item Advocacy page -- meeting or advocacy work in action
    \item Safe Space page -- students or community working together
    \item Volunteer page -- volunteers helping at an event
    \item Internships page -- young professionals or interns collaborating
    \item Speaker page -- speaker presenting to an audience
    \item Media page -- press conference or media interview
\end{itemize}

\section{Testimonials Needed}
\textit{Real quotes with permission to publish. Include name and brief identifier.}
\begin{itemize}
    \item 3 volunteer testimonials -- name, volunteer role, 2-3 sentence quote about experience
    \item 2 intern/alumni testimonials -- name, internship year, current position, quote
    \item 3 donor testimonials -- name, donor type (monthly/annual), quote about why they give
    \item 3 case success stories -- anonymized stories of people CAIR helped (with permission)
\end{itemize}

\vspace{1cm}
\noindent\textbf{Image Requirements:}
\begin{itemize}
    \item All images should be high-resolution (minimum 1200px wide)
    \item Preferred formats: JPG for photos, PNG for graphics with transparency
    \item Hero/banner images should be at least 1920x800 pixels
    \item All photos should be professional quality with good lighting
    \item Ensure you have permission to use all photos on the website
\end{itemize}
